#include <stdio.h>
#include <stdlib.h>
#define MAX 5

void adjacencyMatrix() {
    int matrix[MAX][MAX] = {0};

    matrix[0][1] = matrix[1][0] = 1;
    matrix[0][4] = matrix[4][0] = 1;
    matrix[1][2] = matrix[2][1] = 1;
    matrix[1][3] = matrix[3][1] = 1;
    matrix[1][4] = matrix[4][1] = 1;
    matrix[2][3] = matrix[3][2] = 1;
    matrix[3][4] = matrix[4][3] = 1;

    printf("Adjacency Matrix:\n");
    for(int i=0;i<MAX;i++){
        for(int j=0;j<MAX;j++)
            printf("%d ", matrix[i][j]);
        printf("\n");
    }
}

struct Node {
    int vertex;
    struct Node* next;
};

struct Node* createNode(int v) {
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->vertex = v;
    node->next = NULL;
    return node;
}

void adjacencyList() {
    struct Node* list[MAX] = {NULL};

    void addEdge(int u, int v) {
        struct Node* node = createNode(v);
        node->next = list[u];
        list[u] = node;

        node = createNode(u); 
        node->next = list[v];
        list[v] = node;
    }

    addEdge(0,1); addEdge(0,4);
    addEdge(1,2); addEdge(1,3); addEdge(1,4);
    addEdge(2,3); addEdge(3,4);

    printf("\nAdjacency List:\n");
    for(int i=0;i<MAX;i++){
        printf("%d: ", i);
        struct Node* temp = list[i];
        while(temp){
            printf("%d -> ", temp->vertex);
            temp = temp->next;
        }
        printf("NULL\n");
    }
}

int main() {
    adjacencyMatrix();
    adjacencyList();
    return 0;
}
